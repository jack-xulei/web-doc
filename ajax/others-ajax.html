# jquery 中 Ajax

编写常规的 AJAX 代码并不容易，因为不同的浏览器对 AJAX 的实现并不相同。这意味着您必须编写额外的代码对浏览器进行兼容。不过，jQuery对原生Ajax进行了封装，我们只需要一行简单的代码，就可以实现 AJAX 功能。

 $.ajax({
   url: "https://petstore.swagger.io/v2/store/inventory",
   type: "get",
   data: "userName=admin&userPwd=654321",
   contentType: 'application/json;charset=UTF-8',
   success: function(msg){
      alert(msg );
   },
   error: function (xhr) {
      alert(xhr.status);
   }
});

# jQuery $.get() 方法

$.get() 方法通过 HTTP GET 请求从服务器上请求数据。

$.get(URL,callback);
  • 参数：
    • {string} URL 参数规定您希望请求的 URL
    • {function} callback 参数是请求成功后所执行的函数名

# jQuery $.post() 方法

$.post() 方法通过 HTTP POST 请求从服务器上请求数据。

$.post(URL,data,callback);
  • 参数：
    • {string} URL 参数规定您希望请求的URL
    • {string} data 参数规定连同请求发送的数据
    • {function} callback 可选的 callback 参数是请求成功后所执行的函数名

# Vue axios

Axios 是一个基于 promise 的 HTTP 库，可以用在浏览器和 node.js 中。

import axios from 'axios'
import { MessageBox, Message } from 'element-ui'
import store from '@/store'
import { getToken } from '@/utils/auth'
import { getLanguage, $t } from '@/lang'
import router from '@/router'
let reLogin = false
// create an axios instance
const service = axios.create({
  baseURL: process.env.VUE_APP_BASE_API, // url = base url + request url
  // withCredentials: true, // send cookies when cross-domain requests
  timeout: 10000 // request timeout
})

// request interceptor
service.interceptors.request.use(
  config => {
    // 设置全局loading
    if (config.loading) {
      store.commit('app/ADD_LOADINGLIST', config.url)
    }
    // do something before request is sent
    config.headers['Accept-Language'] = getLanguage()
    const { dep, menuNo } = router.currentRoute.meta
    config.headers['App-Module-Number'] = menuNo || router.resolve({ name: dep }).route.meta.menuNo
    if (store.getters.token) {
      // let each request carry token
      // ['Authentication-Info'] is a custom headers key
      // please modify it according to the actual situation
      config.headers['Authentication-Info'] = getToken()
    }
    return config
  },
  error => {
    // do something with request error
    console.log(error) // for debug
    return Promise.reject(error)
  }
)

// response interceptor
service.interceptors.response.use(
  /**
   * If you want to get http information such as headers or status
   * Please return  response => response
  */

  /**
   * Determine the request status by custom code
   * Here is just an example
   * You can also judge the status by HTTP Status Code
   */
  response => {
    const config = response.config
    const res = response.data

    // 取消全局loading
    if (config.loading) {
      store.commit('app/REMOVE_LOADINGLIST', config.url)
    }
    // if the custom code is not '000001', it is judged as an error.
    if (res.code !== '000001') {
      // 000701: 用户未登陆
      if (res.code === '000701' && !reLogin) {
        reLogin = true
        // to re-login
        MessageBox.confirm($t('login.confirmTip'), $t('login.confirmLogOut'), {
          confirmButtonText: $t('login.confirmBtn1'),
          cancelButtonText: $t('login.confirmBtn2'),
          type: 'warning'
        }).then(() => {
          reLogin = false
          store.dispatch('user/resetToken').then(() => {
            // 修改token失效后重定向到404页面问题
            router.push(`/login?redirect=${router.currentRoute.fullPath}`)
          })
        }).catch(() => {
          reLogin = false
        })
      }
      if (!reLogin) {
        Message({
          message: res.msg || 'Error',
          type: 'error',
          duration: 5 * 1000
        })
      }
      return Promise.reject(new Error(res.msg || 'Error'))
    } else {
      if (config.orgData) {
        return res
      }
      return res.data
    }
  },
  error => {
    console.log('err' + error) // for debug
    const config = error.config
    // 返回错误时也取消全局loading
    if (config.loading) {
      store.commit('app/REMOVE_LOADINGLIST', config.url)
    }
    Message({
      message: error.message,
      type: 'error',
      duration: 5 * 1000
    })
    return Promise.reject(error)
  }
)

export default service
